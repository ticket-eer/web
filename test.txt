    /**
     * Test 5: Transfer timing validation (minimum 15 minutes).
     * A trajet arriving at 14:00 should connect to one departing at 14:15 or later.
     */
    @Test
    void transferTimingValidation() throws Exception {
        LocalDate dateVoyage = LocalDate.now().plusDays(1);

        // Search should respect transfer timing constraints
        MvcResult result = mockMvc.perform(get("/connections/search")
                        .param("villeDepart", "Le Havre")
                        .param("villeArrivee", "Toulouse")
                        .param("dateVoyage", dateVoyage.toString())
                        .header("Authorization", "Bearer " + adminToken))
                .andExpect(status().isOk())
                .andReturn();

        String content = result.getResponse().getContentAsString();
        JsonNode root = objectMapper.readTree(content);

        // For multi-leg connections, check that legs are properly ordered
        JsonNode connections = root.get("content");
        for (JsonNode conn : connections) {
            JsonNode legs = conn.get("legs");
            if (legs.size() > 1) {
                // Multi-leg: verify timing between legs
                for (int i = 0; i < legs.size() - 1; i++) {
                    String arrivalTime = legs.get(i).get("heureArrivee").asText();
                    String nextDepartureTime = legs.get(i + 1).get("heureDepart").asText();
                    // Just verify they exist (actual timing validation is in service)
                    assert arrivalTime != null && nextDepartureTime != null;
                }
            }
        }
    }

    /**
     * Helper to create a trajet via API.
     */
    private void createTrajet(String token, String villeDepart, String villeArrivee,
                              LocalDate dateVoyage, String train, String heureDepart, String heureArrivee) throws Exception {
        String requestBody = String.format(
            "{\"villeDepart\":\"%s\",\"villeArrivee\":\"%s\",\"dateVoyage\":\"%s\",\"train\":\"%s\",\"heureDepart\":\"%s\",\"heureArrivee\":\"%s\",\"itineraireId\":null}",
            villeDepart, villeArrivee, dateVoyage, train, heureDepart, heureArrivee
        );

        mockMvc.perform(post("/trajets")
                        .header("Authorization", "Bearer " + token)
                        .contentType(MediaType.APPLICATION_JSON)
                        .content(requestBody))
                .andExpect(status().isCreated());
    }

    /**
     * Test 13: Admin CRUD for connections with real trajets (create/update/delete).
     */
    @Test
    void adminCrudConnectionsWithRealTrajets() throws Exception {
        LocalDate dateVoyage = LocalDate.now().plusDays(1);

        // First, create some trajets that we'll use in our connections
        Long trajet1Id = createTrajetAndGetId(adminToken, "Paris", "Lyon", dateVoyage, "TGV-400", "09:00", "12:00");
        Long trajet2Id = createTrajetAndGetId(adminToken, "Lyon", "Marseille", dateVoyage, "TGV-401", "13:00", "15:30");
        Long trajet3Id = createTrajetAndGetId(adminToken, "Paris", "Bordeaux", dateVoyage, "TGV-500", "10:00", "13:15");

        // Test 1: Create a single-leg connection
        String singleLegBody = String.format(
                "{\"villeDepart\":\"Paris\",\"villeArrivee\":\"Bordeaux\",\"dateVoyage\":\"%s\",\"trajetIds\":[%d],\"totalDurationMinutes\":195,\"legsCount\":1}",
                dateVoyage, trajet3Id
        );

        String singleLegCreated = mockMvc.perform(post("/admin/connections")
                        .header("Authorization", "Bearer " + adminToken)
                        .contentType(MediaType.APPLICATION_JSON)
                        .content(singleLegBody))
                .andExpect(status().isCreated())
                .andExpect(jsonPath("$.id").isNotEmpty())
                .andExpect(jsonPath("$.villeDepart", equalTo("Paris")))
                .andExpect(jsonPath("$.villeArrivee", equalTo("Bordeaux")))
                .andExpect(jsonPath("$.legsCount", equalTo(1)))
                .andExpect(jsonPath("$.trajetIds", containsString(trajet3Id.toString())))
                .andReturn()
                .getResponse()
                .getContentAsString();

        JsonNode singleLegJson = objectMapper.readTree(singleLegCreated);
        Long singleLegConnectionId = singleLegJson.get("id").asLong();

        // Test 2: Create a multi-leg connection (Paris -> Lyon -> Marseille)
        String multiLegBody = String.format(
                "{\"villeDepart\":\"Paris\",\"villeArrivee\":\"Marseille\",\"dateVoyage\":\"%s\",\"trajetIds\":[%d,%d],\"totalDurationMinutes\":390,\"legsCount\":2}",
                dateVoyage, trajet1Id, trajet2Id
        );

        String multiLegCreated = mockMvc.perform(post("/admin/connections")
                        .header("Authorization", "Bearer " + adminToken)
                        .contentType(MediaType.APPLICATION_JSON)
                        .content(multiLegBody))
                .andExpect(status().isCreated())
                .andExpect(jsonPath("$.id").isNotEmpty())
                .andExpect(jsonPath("$.villeDepart", equalTo("Paris")))
                .andExpect(jsonPath("$.villeArrivee", equalTo("Marseille")))
                .andExpect(jsonPath("$.legsCount", equalTo(2)))
                .andExpect(jsonPath("$.trajetIds", containsString(trajet1Id.toString())))
                .andExpect(jsonPath("$.trajetIds", containsString(trajet2Id.toString())))
                .andReturn()
                .getResponse()
                .getContentAsString();

        JsonNode multiLegJson = objectMapper.readTree(multiLegCreated);
        Long multiLegConnectionId = multiLegJson.get("id").asLong();

        // Test 3: Update the multi-leg connection to add another trajet
        Long trajet4Id = createTrajetAndGetId(adminToken, "Marseille", "Nice", dateVoyage, "TER-100", "16:00", "18:30");
        
        String updateBody = String.format(
                "{\"villeDepart\":\"Paris\",\"villeArrivee\":\"Nice\",\"dateVoyage\":\"%s\",\"trajetIds\":[%d,%d,%d],\"totalDurationMinutes\":540,\"legsCount\":3}",
                dateVoyage, trajet1Id, trajet2Id, trajet4Id
        );

        mockMvc.perform(patch("/admin/connections/" + multiLegConnectionId)
                        .header("Authorization", "Bearer " + adminToken)
                        .contentType(MediaType.APPLICATION_JSON)
                        .content(updateBody))
                .andExpect(status().isOk())
                .andExpect(jsonPath("$.id", equalTo(multiLegConnectionId.intValue())))
                .andExpect(jsonPath("$.villeArrivee", equalTo("Nice")))

                .andExpect(jsonPath("$.trajetIds", containsString(trajet4Id.toString())))

        mockMvc.perform(get("/connections/" + multiLegConnectionId)
                        .header("Authorization", "Bearer " + clientToken))
                .andExpect(status().isOk())
                .andExpect(jsonPath("$.id", equalTo(multiLegConnectionId.intValue())))
                .andExpect(jsonPath("$.villeDepart", equalTo("Paris")))
                .andExpect(jsonPath("$.villeArrivee", equalTo("Nice")))
                .andExpect(jsonPath("$.legsCount", equalTo(3)))
                .andExpect(jsonPath("$.legs", hasSize(3)));

        // Test 5: Delete both connections
        mockMvc.perform(delete("/admin/connections/" + singleLegConnectionId)
                        .header("Authorization", "Bearer " + adminToken))
                .andExpect(status().isNoContent());

        mockMvc.perform(delete("/admin/connections/" + multiLegConnectionId)
                        .header("Authorization", "Bearer " + adminToken))
                .andExpect(status().isNoContent());

        // Test 6: Ensure they are gone
        mockMvc.perform(get("/connections/" + singleLegConnectionId)
                        .header("Authorization", "Bearer " + clientToken))
                .andExpect(status().isNotFound());

        mockMvc.perform(get("/connections/" + multiLegConnectionId)
                        .header("Authorization", "Bearer " + clientToken))
                .andExpect(status().isNotFound());
    }

    /**
     * Test 14: Test connection search with multiple trajets to verify functionality works.
     */
    @Test
    void connectionSearchWithMultipleTrajetsWorks() throws Exception {
        LocalDate dateVoyage = LocalDate.now().plusDays(2);

        // Create a complex route: Lille -> Paris -> Lyon -> Marseille -> Nice
        createTrajetAndGetId(adminToken, "Lille", "Paris", dateVoyage, "TGV-600", "07:00", "08:00");
        createTrajetAndGetId(adminToken, "Paris", "Lyon", dateVoyage, "TGV-601", "09:00", "12:00");
        createTrajetAndGetId(adminToken, "Lyon", "Marseille", dateVoyage, "TGV-602", "13:00", "15:30");
        createTrajetAndGetId(adminToken, "Marseille", "Nice", dateVoyage, "TER-200", "16:00", "18:30");

        // Also create some alternative routes
        createTrajetAndGetId(adminToken, "Lille", "Lyon", dateVoyage, "TGV-700", "08:30", "13:00");
        createTrajetAndGetId(adminToken, "Paris", "Marseille", dateVoyage, "TGV-800", "10:00", "15:00");

        // Test 1: Search for direct route (should find it)
        MvcResult directResult = mockMvc.perform(get("/connections/search")
                        .param("villeDepart", "Lille")
                        .param("villeArrivee", "Paris")
                        .param("dateVoyage", dateVoyage.toString())
                        .header("Authorization", "Bearer " + clientToken))
                .andExpect(status().isOk())
                .andExpect(jsonPath("$.totalCount", greaterThanOrEqualTo(1)))
                .andReturn();

        String directContent = directResult.getResponse().getContentAsString();
        JsonNode directRoot = objectMapper.readTree(directContent);
        
        // Verify we found at least one direct connection
        assert directRoot.get("totalCount").asInt() >= 1 : "Should find direct Lille->Paris connection";
        
        // Check that the first connection has 1 leg
        JsonNode firstDirectConn = directRoot.get("content").get(0);
        assert firstDirectConn.get("legsCount").asInt() == 1 : "Direct connection should have 1 leg";

        // Test 2: Search for multi-leg route (Lille -> Nice)
        MvcResult multiLegResult = mockMvc.perform(get("/connections/search")
                        .param("villeDepart", "Lille")
                        .param("villeArrivee", "Nice")
                        .param("dateVoyage", dateVoyage.toString())
                        .header("Authorization", "Bearer " + clientToken))
                .andExpect(status().isOk())
                .andReturn();

        String multiLegContent = multiLegResult.getResponse().getContentAsString();
        JsonNode multiLegRoot = objectMapper.readTree(multiLegContent);
        
        // Should find at least one multi-leg connection
        int totalConnections = multiLegRoot.get("totalCount").asInt();
        assert totalConnections >= 1 : "Should find at least one Lille->Nice connection";

        // Verify that we have connections with multiple legs
        JsonNode connections = multiLegRoot.get("content");
        boolean foundMultiLeg = false;
        for (JsonNode conn : connections) {
            if (conn.get("legsCount").asInt() > 1) {
                foundMultiLeg = true;
                // Verify the connection has proper structure
                assert conn.get("legs").isArray() : "Connection should have legs array";
                assert conn.get("legs").size() == conn.get("legsCount").asInt() : "Legs array size should match legsCount";
                break;
            }
        }
        assert foundMultiLeg : "Should find at least one multi-leg connection";

        // Test 3: Search for intermediate route (Paris -> Marseille)
        mockMvc.perform(get("/connections/search")
                        .param("villeDepart", "Paris")
                        .param("villeArrivee", "Marseille")
                        .param("dateVoyage", dateVoyage.toString())
                        .header("Authorization", "Bearer " + clientToken))
                .andExpect(status().isOk())
                .andExpect(jsonPath("$.totalCount", greaterThanOrEqualTo(1)))
                .andExpect(jsonPath("$.content[0].villeDepart", equalToIgnoringCase("Paris")))
                .andExpect(jsonPath("$.content[0].villeArrivee", equalToIgnoringCase("Marseille")));
    }

                .andExpect(jsonPath("$.trajetIds", containsString(trajet3Id.toString())))
                .andExpect(jsonPath("$.trajetIds", not(containsString(trajet4Id.toString()))));

        // Verify the updated connection
        mockMvc.perform(get("/connections/" + complexConnectionId)
                        .header("Authorization", "Bearer " + clientToken))
                .andExpect(status().isOk())
                .andExpect(jsonPath("$.legsCount", equalTo(3)))
                .andExpect(jsonPath("$.legs", hasSize(3)))
                .andExpect(jsonPath("$.villeArrivee", equalTo("Bordeaux")));

        // Clean up
        mockMvc.perform(delete("/admin/connections/" + complexConnectionId)
                        .header("Authorization", "Bearer " + adminToken))
                .andExpect(status().isNoContent());
    }

    /**
     * Helper method to create a trajet and return its ID.
     */
    private Long createTrajetAndGetId(String token, String villeDepart, String villeArrivee,
                                      LocalDate dateVoyage, String train, String heureDepart, String heureArrivee) throws Exception {
        String requestBody = String.format(
            "{\"villeDepart\":\"%s\",\"villeArrivee\":\"%s\",\"dateVoyage\":\"%s\",\"train\":\"%s\",\"heureDepart\":\"%s\",\"heureArrivee\":\"%s\",\"itineraireId\":null}",
            villeDepart, villeArrivee, dateVoyage, train, heureDepart, heureArrivee
        );

        MvcResult result = mockMvc.perform(post("/trajets")
                        .header("Authorization", "Bearer " + token)
                        .contentType(MediaType.APPLICATION_JSON)
                        .content(requestBody))
                .andExpect(status().isCreated())
                .andReturn();

        String responseContent = result.getResponse().getContentAsString();
        JsonNode responseJson = objectMapper.readTree(responseContent);
        return responseJson.get("id").asLong();
    }
